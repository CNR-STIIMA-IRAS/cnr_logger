60 class LOG4CXX_EXPORT ColorPatternLayout : public log4cxx::PatternLayout
61 {
62 public:
63  DECLARE_LOG4CXX_OBJECT(ColorPatternLayout)
64  BEGIN_LOG4CXX_CAST_MAP()
65  LOG4CXX_CAST_ENTRY(ColorPatternLayout)
66  LOG4CXX_CAST_ENTRY_CHAIN(Layout)
67  END_LOG4CXX_CAST_MAP()
68 
69  ColorPatternLayout() : log4cxx::PatternLayout() {}
70  ColorPatternLayout(const log4cxx::LogString &s): log4cxx::PatternLayout(s) {}
71  virtual void format(log4cxx::LogString &output,
72  const log4cxx::spi::LoggingEventPtr &event,
73  log4cxx::helpers::Pool &pool) const override
74  {
75  log4cxx::LogString tmp;
76  log4cxx::PatternLayout::format(tmp, event, pool);
77  log4cxx::LevelPtr lvl = event->getLevel();
78  switch (lvl->toInt())
79  {
80  case log4cxx::Level::FATAL_INT:
81  output.append("\u001b[0;41m"); // red BG
82  break;
83  case log4cxx::Level::ERROR_INT:
84  output.append("\u001b[0;31m"); // red FG
85  break;
86  case log4cxx::Level::WARN_INT:
87  output.append("\u001b[0;33m"); // Yellow FG
88  break;
89  case log4cxx::Level::INFO_INT:
90  output.append("\u001b[1m"); // Bright
91  break;
92  case log4cxx::Level::DEBUG_INT:
93  output.append("\u001b[1;32m"); // Green FG
94  break;
95  case log4cxx::Level::TRACE_INT:
96  output.append("\u001b[0;34m"); // Black FG
97  break;
98  default:
99  break;
100  }
101  output.append(tmp);
102  output.append("\u001b[m");
103  }
104 };
105 LOG4CXX_PTR_DEF(ColorPatternLayout);
